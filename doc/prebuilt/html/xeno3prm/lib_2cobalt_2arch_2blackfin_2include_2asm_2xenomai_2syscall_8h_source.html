25 #define __emit_syscall0(syscode, ...) \
26 ({ \
27  long __res; \
28  __asm__ __volatile__ ( \
29  "excpt 0;\n\t" \
30  : "=q0" (__res) \
31  : "qA" (syscode), \
32  ##__VA_ARGS__ \
33  : "CC", "memory"); \
34  __res; \
35 })
